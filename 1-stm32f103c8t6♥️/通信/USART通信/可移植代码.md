void UART1_Msp_Init(){
	// 使能GPIOA时钟
	__HAL_RCC_GPIOA_CLK_ENABLE();
	// 使能USART1时钟
	__HAL_RCC_USART1_CLK_ENABLE();
	
	GPIO_InitTypeDef GPIO_InitStruct = {0};
	
	// 2. 配置PA9(TX）
	GPIO_InitStruct.Pin = GPIO_PIN_9;
	GPIO_InitStruct.Mode = GPIO_MODE_AF_PP; // 复用推挽输出
	GPIO_InitStruct.Pull = GPIO_NOPULL;
	GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_HIGH; // 高速模式
	HAL_GPIO_Init(GPIOA, &GPIO_InitStruct);
	
	// 配置PA10(RX)
	GPIO_InitStruct.Pin = GPIO_PIN_10;
	GPIO_InitStruct.Mode = GPIO_MODE_INPUT; // 浮空输入模式
	GPIO_InitStruct.Pull = GPIO_NOPULL;
	HAL_GPIO_Init(GPIOA, &GPIO_InitStruct);
}

UART_HandleTypeDef UART1_Init(int buadRate){
	UART1_Msp_Init();
	
	UART_HandleTypeDef huart1;
	
	huart1.Instance = USART1; // 选择USART1
	huart1.Init.BaudRate = buadRate; // 波特率
	huart1.Init.WordLength = UART_WORDLENGTH_8B; // 8位数据
	huart1.Init.StopBits = UART_STOPBITS_1; // 1位停止位
	huart1.Init.Parity = UART_PARITY_NONE; // 无校验位
	huart1.Init.Mode = UART_MODE_TX_RX; // 收发模式
	huart1.Init.HwFlowCtl = UART_HWCONTROL_NONE; // 无硬件流控
	huart1.Init.OverSampling = UART_OVERSAMPLING_16; // 16倍过采样
	
	HAL_UART_Init(&huart1); // 应用配置
	
	return huart1;
}
