# 读取整个文件

```cpp
#include <iostream>
#include <fstream>
#include <sstream>
#include <string>

std::string readFile(const std::string& path) {
    std::ifstream file(path);
    std::stringstream buffer;
    buffer << file.rdbuf();
    return buffer.str();
}

int main() {
    std::string content = readFile("example.txt");
    std::cout << "File content:\n" << content << std::endl;
    return 0;
}
```
